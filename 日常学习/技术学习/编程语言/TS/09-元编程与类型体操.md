# TypeScript 元编程与类型体操

## 详细描述

TypeScript 的元编程分为两个层面：**类型层面的元编程**（类型体操，编译时，零运行时开销）和**运行时层面的元编程**（装饰器、reflect-metadata、Proxy）。两者互补，构成了 TS 强大的抽象能力。

**类型体操**（Type-level programming）是 TS 最独特的魅力。通过组合条件类型（`T extends U ? X : Y`）、映射类型（`{ [K in keyof T]: ... }`）、递归类型、`infer` 和模板字面量类型，可以在类型层面实现极其复杂的计算和变换。这种"类型即程序"的能力是 TS 区别于几乎所有其他类型系统的特点。

**条件类型**的分配律（Distributive Conditional Types）是实现联合类型变换的基础：当 `T` 是联合类型时，`T extends U ? X : Y` 自动分布在联合的每个成员上。利用这个特性，`Exclude`、`Extract`、`NonNullable` 等内置工具类型只需一行代码。

**映射类型**配合 `as` 子句（Key Remapping，TS 4.1）可以实现键的过滤、重命名、条件映射：`` { [K in keyof T as K extends string ? `${Prefix}${K}` : never]: T[K] } ``。这让类型变换的表达力达到新的高度。

**递归条件类型**允许类型引用自身，实现深度变换：`DeepPartial<T>` 递归地将所有嵌套属性的 `?` 标记添加。递归深度有编译器限制（通常 50 层），需注意性能。

**装饰器**（Decorator）是 TS 运行时元编程的主要手段。类装饰器、方法装饰器、属性装饰器、参数装饰器可以在定义时修改或增强目标的行为。`reflect-metadata` 库为装饰器提供了运行时类型存储能力，类似 C# 的 `[Attribute]` 机制。Angular 和 NestJS 等框架大量使用装饰器进行依赖注入和路由注册。

**Proxy**（ES2015）是运行时拦截的利器。可以拦截对象的所有基本操作（get、set、has、ownKeys 等），配合 `Reflect` 进行默认行为转发。Vue 3 的响应式系统完全基于 Proxy 构建。

## 代码实例

```typescript
// ========== 1. 条件类型与 infer ==========

// 根据条件变换类型
type IsNever<T> = [T] extends [never] ? true : false;  // 防止分配律
type A = IsNever<never>;  // true（正确！）

// 深度 infer：提取函数参数元组
type Parameters2<T> = T extends (...args: infer P) => any ? P : never;
type FnParams = Parameters2<(a: number, b: string) => void>;  // [number, string]

// 提取 Promise 内部类型
type Awaited<T> = T extends Promise<infer U> ? Awaited<U> : T;
type P1 = Awaited<Promise<string>>;          // string
type P2 = Awaited<Promise<Promise<number>>>; // number

// ========== 2. 映射类型进阶 ==========

// 键重映射（TS 4.1+）
type Getters<T> = {
    [K in keyof T as `get${Capitalize<string & K>}`]: () => T[K];
};

interface Person {
    name: string;
    age: number;
}
type PersonGetters = Getters<Person>;
// { getName: () => string; getAge: () => number }

// 条件键过滤
type OnlyStringKeys<T> = {
    [K in keyof T as T[K] extends string ? K : never]: T[K];
};

type StringProps = OnlyStringKeys<{ name: string; age: number; email: string }>;
// { name: string; email: string }

// 属性修饰符变换
type MakeWritable<T> = { -readonly [K in keyof T]-?: T[K] };

// ========== 3. 递归类型 ==========

// DeepPartial：递归标记所有属性为可选
type DeepPartial<T> = T extends object
    ? { [K in keyof T]?: DeepPartial<T[K]> }
    : T;

type PartialPerson = DeepPartial<{
    user: {
        name: string;
        address: { street: string; city: string };
    };
}>;
// { user?: { name?: string; address?: { street?: string; city?: string } } }

// DeepReadonly
type DeepReadonly<T> = T extends object
    ? { readonly [K in keyof T]: DeepReadonly<T[K]> }
    : T;

// Path 类型：递归提取对象的所有路径
type Path<T, Prefix extends string = ""> = {
    [K in keyof T]: T[K] extends object
        ? `${Prefix & string}${K & string}` | Path<T[K], `${Prefix & string}${K & string}.`>
        : `${Prefix & string}${K & string}`;
}[keyof T];

type PersonPaths = Path<{ user: { name: string; address: { city: string } } }>;
// "user" | "user.name" | "user.address" | "user.address.city"

// ========== 4. 模板字面量类型体操 ==========

// 将联合类型转为字符串排列
type Permutation<T extends string, U extends string = T> =
    [T] extends [never]
        ? ""
        : T extends infer V extends string
            ? `${V}${Permutation<Exclude<U, V>>}`
            : never;

type P = Permutation<"a" | "b" | "c">;
// "abc" | "acb" | "bac" | "bca" | "cab" | "cba"

// 实用的字符串变换
type CamelCase<S extends string> =
    S extends `${infer P1}_${infer P2}`
        ? `${Lowercase<P1>}${Capitalize<CamelCase<P2>>}`
        : Lowercase<S>;

type CamelCased = CamelCase<"USER_FIRST_NAME">;  // "userFirstName"

// ========== 5. 装饰器（运行时元编程） ==========

// 类装饰器
function Entity(tableName: string) {
    return function <T extends { new (...args: any[]): {} }>(target: T) {
        Reflect.defineMetadata("entity:table", tableName, target);
        return target;
    };
}

// 属性装饰器
function Column(type: string, options?: { primary?: boolean }) {
    return function (target: any, propertyKey: string) {
        const columns = Reflect.getMetadata("entity:columns", target.constructor) ?? [];
        columns.push({ property: propertyKey, type, ...options });
        Reflect.defineMetadata("entity:columns", columns, target.constructor);
    };
}

// 方法装饰器：测量执行时间
function Measure(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value;
    descriptor.value = function (...args: any[]) {
        const start = performance.now();
        const result = original.apply(this, args);
        const end = performance.now();
        console.log(`${propertyKey} took ${(end - start).toFixed(2)}ms`);
        return result;
    };
    return descriptor;
}

// 使用装饰器
@Entity("users")
class User {
    @Column("number", { primary: true })
    id!: number;

    @Column("string")
    name!: string;

    @Measure
    longRunningTask(): void {
        // 模拟耗时计算
        let x = 0;
        for (let i = 0; i < 1000000; i++) x += Math.sqrt(i);
    }
}

// ========== 6. Proxy 运行时拦截 ==========

// 访问日志代理
function createLoggingProxy<T extends object>(target: T, name: string): T {
    return new Proxy(target, {
        get(obj, prop) {
            console.log(`[${name}] GET ${String(prop)}`);
            return Reflect.get(obj, prop);
        },
        set(obj, prop, value) {
            console.log(`[${name}] SET ${String(prop)} = ${JSON.stringify(value)}`);
            return Reflect.set(obj, prop, value);
        },
    });
}

// 备忘录代理：自动缓存函数结果
function memoize<Args extends unknown[], Return>(
    fn: (...args: Args) => Return,
): (...args: Args) => Return {
    const cache = new Map<string, Return>();
    return new Proxy(fn, {
        apply(target, thisArg, args) {
            const key = JSON.stringify(args);
            if (cache.has(key)) {
                console.log("Cache hit:", key);
                return cache.get(key)!;
            }
            const result = Reflect.apply(target, thisArg, args);
            cache.set(key, result);
            return result;
        },
    });
}
```

## 推荐书籍

1. **《Programming TypeScript》** — Boris Cherny
   第 7 章（Advanced Types）和第 9 章（Namespaces.Modules）有深入的映射类型和条件类型讲解。

2. **《Effective TypeScript》** — Dan Vanderkam
   条款 50-55 涵盖了装饰器、条件类型、映射类型和模板字面量类型的最佳实践。

3. **《TypeScript Cookbook》** — Stefan Baumgartner
   包含大量实际项目中类型体操的案例，以及装饰器在 NestJS 中的高级应用。

4. **Type Challenges**（在线开源项目）— https://github.com/type-challenges/type-challenges
   通过 100+ 道类型体操练习题，从 Easy 到 Extreme 系统训练类型编程能力。

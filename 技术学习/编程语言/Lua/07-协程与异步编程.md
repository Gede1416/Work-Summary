# Lua 协程与异步编程

## 详细描述

Lua 的并发模型非常独特——**没有原生线程**，但提供了**非抢占式协程**（Coroutine）。这一设计使得 Lua 在嵌入式场景中非常安全（无锁、无竞态），同时也为异步 I/O 调度提供了轻量级的构建基础。

**协程基础**：Lua 协程通过 `coroutine.create(f)` 创建，返回一个协程对象（类型为 `thread`）。`coroutine.resume(co, ...)` 开始或继续执行协程，传入的参数成为 `coroutine.yield(...)` 的返回值。`coroutine.yield(...)` 暂停协程的执行，将控制权交还给调用 `resume` 的一方，同时传入的参数成为 `resume` 的返回值。协程的状态有：`suspended`（挂起）、`running`（运行中）、`normal`（激活了其他协程）、`dead`（已结束）。

**非抢占式调度**意味着协程必须主动 `yield` 才能让出执行权——没有时钟中断抢占当前协程。这是 Lua 协程与操作系统线程的根本区别。优点是：不需要考虑竞态条件（在 yield 之间的代码是原子的），不需要锁。缺点是：一个死循环会阻塞所有其他协程。

**生产者-消费者模式**是协程的经典应用。生产者协程生成数据并通过 yield 传递，消费者通过 resume 获取数据。这实现了拉取式（pull-based）的数据流——消费者驱动生产者生成数据。

**异步 I/O 调度**是协程在实战中的主要应用。以 socket 操作为例：当协程尝试对未就绪的 socket 进行 read 时，将 socket 和协程注册到事件循环（如 select/poll），然后 yield 让出控制权。当 socket 可读时，事件循环 resume 对应的协程。这就实现了"看起来同步"的异步 I/O 代码——每个连接一个协程，代码是线性的，但底层是完全异步调度的。

**LuaSocket** 是 Lua 最常用的网络库，配合协程和调度器（如 Copas），可以实现高并发的 TCP/UDP 服务。OpenResty（Nginx + LuaJIT）的 `ngx.sleep`、`ngx.socket.tcp` 等都是协程在异步 I/O 中的成熟应用。

**wrap 函数**：`coroutine.wrap(f)` 返回一个函数，调用它就相当于 `resume`，但以函数调用而非 `resume` 的方式使用。更简单，但无法检查错误。

## 代码实例

```lua
-- ========== 1. 协程基础 ==========

-- 创建协程
local co = coroutine.create(function()
    print("协程开始")
    local received = coroutine.yield("第一次 yield")
    print("收到:", received)
    return "完成"
end)

print("状态:", coroutine.status(co))  --> "suspended"

-- 第一次 resume
local ok, yielded = coroutine.resume(co)
print("resume 返回:", ok, yielded)     --> true, "第一次 yield"
print("状态:", coroutine.status(co))  --> "suspended"

-- 第二次 resume（传值给 yield）
local ok2, result = coroutine.resume(co, "来自外部的消息")
print("resume 返回:", ok2, result)    --> true, "完成"
print("状态:", coroutine.status(co))  --> "dead"

-- ========== 2. 生产者-消费者 ==========

-- 生成器：产生斐波那契数列
local function fibonacciGenerator()
    local a, b = 0, 1
    while true do
        coroutine.yield(a)
        a, b = b, a + b
    end
end

local function printFibonacci(limit)
    local co = coroutine.create(fibonacciGenerator)
    for i = 1, limit do
        local ok, value = coroutine.resume(co)
        if not ok then break end
        print(string.format("Fib(%d) = %d", i, value))
    end
end

printFibonacci(10)
-- Fib(1) = 0, Fib(2) = 1, Fib(3) = 1, Fib(4) = 2, ...

-- 通用迭代器包装
local function producer(genFunc)
    local co = coroutine.create(genFunc)
    return function()
        local ok, value = coroutine.resume(co)
        if ok and coroutine.status(co) ~= "dead" then
            return value
        end
        return nil
    end
end

-- 使用协程迭代器
for value in producer(function()
    for i = 1, 5 do
        coroutine.yield(i * 2)
    end
end) do
    print("迭代:", value)
end
-- 迭代: 2, 4, 6, 8, 10

-- ========== 3. coroutine.wrap 简化用法 ==========

local function createRange(start, stop, step)
    step = step or 1
    return coroutine.wrap(function()
        for i = start, stop, step do
            coroutine.yield(i)
        end
    end)
end

for i in createRange(0, 30, 10) do
    print(i)  --> 0, 10, 20, 30
end

-- ========== 4. 异步 I/O 调度器（概念示例） ==========

-- 模拟一个简易事件循环
local Scheduler = {}
Scheduler.__index = Scheduler

function Scheduler:new()
    return setmetatable({
        waiting = {},  -- { [socket] = coroutine }
        ready = {},    -- 就绪的 socket 列表
    }, self)
end

-- 注册等待读取的协程
function Scheduler:waitRead(socket, co)
    self.waiting[socket] = co
end

-- 模拟 I/O 事件通知
function Scheduler:notifyReady(socket)
    self.ready[#self.ready + 1] = socket
end

-- 运行事件循环
function Scheduler:run()
    while next(self.waiting) do
        -- 模拟 select/poll：检查哪些 socket 就绪
        -- 在实际场景中，这里会调用 select()、poll() 或 epoll()
        
        for _, sock in ipairs(self.ready) do
            local co = self.waiting[sock]
            if co and coroutine.status(co) == "suspended" then
                self.waiting[sock] = nil
                coroutine.resume(co, sock)
            end
        end
        
        -- 清空就绪列表（简化的单次处理）
        for i = #self.ready, 1, -1 do
            self.ready[i] = nil
        end
    end
end

-- 模拟异步读取
local function asyncRead(scheduler, socket)
    local co = coroutine.running()
    scheduler:waitRead(socket, co)
    local readySocket = coroutine.yield()  -- 挂起等待数据
    return "data from " .. readySocket
end

-- 使用示例
local sched = Scheduler:new()

-- 启动多个"连接"的协程
local co1 = coroutine.create(function()
    local data = asyncRead(sched, "socket-1")
    print("协程1 收到:", data)
    local data2 = asyncRead(sched, "socket-1")
    print("协程1 再次收到:", data2)
end)

local co2 = coroutine.create(function()
    local data = asyncRead(sched, "socket-2")
    print("协程2 收到:", data)
end)

coroutine.resume(co1)
coroutine.resume(co2)

-- 模拟数据到达
sched:notifyReady("socket-1")
sched:notifyReady("socket-2")
sched:run()

-- ========== 5. 多任务协作 ==========
-- 模拟多个协程轮流执行

local function task(name, duration)
    for i = 1, duration do
        print(string.format("%s: 步骤 %d", name, i))
        coroutine.yield()  -- 主动让出 CPU
    end
    print(name .. ": 完成！")
end

local function cooperativeScheduler(tasks)
    local coroutines = {}
    for _, t in ipairs(tasks) do
        local co = coroutine.create(function()
            task(t.name, t.duration)
        end)
        table.insert(coroutines, co)
        coroutine.resume(co)  -- 启动所有协程（执行到第一个 yield）
    end
    
    -- 轮询调度
    local allDone = false
    while not allDone do
        allDone = true
        for _, co in ipairs(coroutines) do
            if coroutine.status(co) ~= "dead" then
                allDone = false
                coroutine.resume(co)
            end
        end
    end
end

cooperativeScheduler({
    { name = "Task A", duration = 3 },
    { name = "Task B", duration = 2 },
    { name = "Task C", duration = 4 },
})
```

## 推荐书籍

1. **《Programming in Lua》（第4版）** — Roberto Ierusalimschy
   第 9 章（Coroutines）和第 24 章（Coroutines 实战）是 Lua 协程的权威讲解，涵盖了从基础到调度器实现的全部内容。

2. **《OpenResty Best Practices》**（在线，开源社区）
   详述了 Nginx + LuaJIT 中协程在 ngx.sleep、ngx.location.capture 等 API 中的实际运作，是协程在生产环境中应用的最佳参考。

3. **《Lua Programming Gems》** — L.H. de Figueiredo
   第 6 章专门讨论了协程在异步 I/O 和游戏脚本中的应用模型和设计模式。

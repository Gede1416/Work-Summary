# Lua 调试与性能分析

## 详细描述

Lua 的调试和性能分析工具相对简单，但足够满足大多数场景。主要依赖内建的 `debug` 库、宿主环境的调试器，以及手动性能测量。

**debug 库**是 Lua 最基础的调试工具。`debug.traceback(msg)` 生成调用栈字符串，是最常用的错误诊断方式。`debug.getinfo(fn)` 获取函数的源文件、行号、参数数等信息。`debug.getlocal(level, index)` 和 `debug.setlocal()` 可以检查和修改局部变量。`debug.sethook(hook, mask, count)` 设置钩子函数，在特定事件（函数调用、返回、每行执行）时触发。

**宿主环境调试器**是 Lua 嵌入式场景中的主要调试方式。游戏引擎（Unity、Unreal、Cocos2d-x）通常内建 Lua 调试器，提供断点、变量监视、表达式求值等标准调试功能。Neovim 使用内置的 Lua 运行时调试。OpenResty 有专门的命令行调试工具。

**Lua 调试器**（如 MobDebug、lldebugger）是基于 debug 库的远程调试器。它们通过 debug hook 和 socket 连接提供完整的 IDE 体验（断点、单步、监视）。

**性能分析工具有限**——Lua 没有内置的 profiler，主要靠：
- `os.clock()` 手动测量执行时间
- debug hook 实现 profiling（统计每个函数的调用次数和时间）
- LuaJIT 的 `-jp` 标志（`-jp=pf` 输出函数时间统计，`-jp=fl` 输出行级统计）
- 第三方库（如 luaprofiler、profi.lua）

**手动跟踪**是最实用的方法：在关键代码路径上使用 `local start = os.clock()` 和 `print(os.clock() - start)` 测量耗时。

## 代码实例

```lua
-- ========== 1. 手动性能测量 ==========

-- 精确计时
local function measure(label, func, ...)
    local start = os.clock()
    local results = {func(...)}
    local elapsed = os.clock() - start
    print(string.format("%s: %.4f ms", label, elapsed * 1000))
    return table.unpack(results)
end

-- 带重复的基准测试
local function benchmark(label, func, iterations)
    iterations = iterations or 100000
    -- 预热
    for i = 1, 100 do func() end
    
    local start = os.clock()
    for i = 1, iterations do func() end
    local elapsed = os.clock() - start
    print(string.format("[%s] %d iterations: %.2f µs/op",
        label, iterations, (elapsed / iterations) * 1e6))
end

-- 示例：对比字符串拼接方式
local items = {}
for i = 1, 1000 do table.insert(items, tostring(i)) end

benchmark("table.concat", function()
    local s = table.concat(items, ",")
end)

benchmark(".. 拼接", function()
    local s = ""
    for _, item in ipairs(items) do
        s = s .. item .. ","
    end
end)
-- table.concat 通常快 10-50×

-- ========== 2. debug 库诊断 ==========

-- 详细的错误处理
local function detailedError(handler, ...)
    local results = {xpcall(handler, function(err)
        local trace = debug.traceback(tostring(err), 2)
        print("[ERROR] " .. trace)
        -- 在这里可以将错误写入日志文件
    end, ...)}
    
    return table.unpack(results)
end

-- 查看调用栈
local function showCallStack()
    local level = 1
    while true do
        local info = debug.getinfo(level, "nSl")
        if not info then break end
        
        local name = info.name or "(anonymous)"
        local source = info.source or "?"
        local line = info.currentline or info.linedefined or 0
        
        print(string.format("  #%d: %s at %s:%d", level, name, source, line))
        level = level + 1
    end
end

-- ========== 3. 简单 Profiler（基于 debug hook） ==========

local Profiler = {}
Profiler.__index = Profiler

function Profiler:new()
    local p = {
        data = {},       -- { [func_name] = { calls, total_time, self_time } }
        callStack = {},   -- 当前调用栈
        startTime = nil,
    }
    setmetatable(p, self)
    return p
end

function Profiler:start()
    self.data = {}
    self.callStack = {}
    self.startTime = os.clock()
    
    debug.sethook(function(event)
        local info = debug.getinfo(2, "nS")
        local funcName = info.name or info.source or "?"
        
        if event == "call" then
            if not self.data[funcName] then
                self.data[funcName] = {calls = 0, total = 0}
            end
            self.data[funcName].calls = self.data[funcName].calls + 1
            table.insert(self.callStack, {name = funcName, time = os.clock()})
        elseif event == "return" then
            local call = table.remove(self.callStack)
            if call and self.data[call.name] then
                local elapsed = os.clock() - call.time
                self.data[call.name].total = self.data[call.name].total + elapsed
            end
        end
    end, "cr")
end

function Profiler:stop()
    debug.sethook()  -- 清除 hook
end

function Profiler:report()
    print("\n=== Profiler Report ===")
    local sorted = {}
    for name, info in pairs(self.data) do
        table.insert(sorted, {name = name, calls = info.calls, total = info.total})
    end
    table.sort(sorted, function(a, b) return a.total > b.total end)
    
    for i, entry in ipairs(sorted) do
        if i > 20 then break end  -- 只显示前 20
        print(string.format("  %2d. %-40s calls: %5d  time: %8.2f ms",
            i, entry.name:sub(1, 40), entry.calls, entry.total * 1000))
    end
end

-- 使用 Profiler
local function slowFunction()
    for i = 1, 50000 do
        local x = math.sin(i) * math.cos(i)
    end
end

local function fastFunction()
    local x = 1 + 1
end

local function main()
    for i = 1, 10 do slowFunction() end
    for i = 1, 1000 do fastFunction() end
end

local profiler = Profiler:new()
profiler:start()
main()
profiler:stop()
profiler:report()

-- ========== 4. LuaJIT 性能分析 ==========
-- 命令行选项（无需修改代码）
-- luajit -jp=pf myapp.lua     -- 统计函数时间

-- ========== 5. 内存分析技巧 ==========
-- 跟踪对象创建
local allocatedObjects = setmetatable({}, { __mode = "k" })

local function createTrackedObject(name)
    local obj = { name = name }
    allocatedObjects[obj] = os.time()
    return obj
end

-- 检查 GC 状态
print("Memory before:", collectgarbage("count"), "KB")
local objects = {}
for i = 1, 100000 do
    table.insert(objects, { data = string.rep("x", 100) })
end
print("Memory after allocation:", collectgarbage("count"), "KB")
objects = {}  -- 释放引用
collectgarbage("collect")
print("Memory after GC:", collectgarbage("count"), "KB")
```

## 推荐书籍

1. **《Programming in Lua》（第4版）** — Roberto Ierusalimschy
   第 23 章（The Debug Library）全面介绍了 debug 库的所有 API 和调试技术。

2. **LuaJIT 官方文档**（luajit.org）
   `-jv`、`-jp` 标志的详细说明和 profiling 输出的解读方法。

3. **《Lua Programming Gems》** — L.H. de Figueiredo
   第 8 章讨论了游戏环境中的 Lua 调试技术和性能优化策略。

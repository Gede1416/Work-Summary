/// <summary>刷新两条小助手显隐与内容（layout：上 assistant2 缺货，下 assistant 建造）</summary>
private void RefreshAssistantHints()
{
    bool showBuild = MainAssistantHintUtil.HasBuildAssistantType1Hint();
    if (showBuild)
    {
        txAssistant.SetText(MainAssistantHintUtil.GetAssistantHintText(1, 0));
    }

    bool showProd = MainAssistantHintUtil.TryGetProductionStockOutHint(out int prodAssistantType, out uint productId,
        out uint pageJump, out CfgEnum.BuildingType prodBuildingType);
    prodAssistantPageJumpId = showProd ? pageJump : 0;
    if (showProd)
    {
        txAssistant2.SetText(MainAssistantHintUtil.GetAssistantHintText(prodAssistantType, productId));
        var itemCfg = Cfg.Ins.item.get(productId);
        if (itemCfg != null && !string.IsNullOrEmpty(itemCfg.Icon))
        {
            imgIconAssistant2.sprite = ImageUtil.GetInstance().GetSpriteByPath(itemCfg.Icon);
            imgIconAssistant2.gameObject.SetActive(true);
        }
        else
        {
            imgIconAssistant2.gameObject.SetActive(false);
        }
    }
    else
    {
        imgIconAssistant2.gameObject.SetActive(false);
    }

    btnAssistant.gameObject.SetActive(showBuild);
    btnAssistant2.gameObject.SetActive(showProd);
}

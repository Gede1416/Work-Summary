# AssetBundle 基础

## AssetBundle 打包
1. 设置备打包文件
``` cs
    string abName // 打包后名称
    string assetPath = AssetDatabase.GetAssetPath(Selection.activeObject); // 获取选中对象路径
    AssetImporter importer = AssetImporter.GetAtPath(assetPath); // 获取导入器 .mate
    importer.assetBundleName = abName; // 设置打包名称
    importer.assetBundleVariant = ""; // 设置打包后后缀

    // 保存操作
    EditorUtility.SetDirty(importer); // 保存导入器
```

2. 进行打包
``` cs
    AssetBundleManifest manifest = BuildPipeline.BuildAssetBundles(
        string, // 打包输出路径
        BuildAssetBundleOptions, // 打包选项
        BuildTarget // 打包平台
    );

    manifest.GetAllAssetBundles(); // 获取所有打包文件
    manifest.GetAllDependencies(bundleName); // 获取依赖文件
    
    // 保存操作
    AssetDatabase.Refresh(); // 刷新资源

    //弹窗
    EditorUtility.DisplayDialog( 
        "打包完成",
        $"共 {allBundles.Length} 个 AssetBundle 已输出到:\n{outputDir}",
        "确定"
    );
```

3. 清理操作
``` cs
    string[] allABNames = AssetDatabase.GetAllAssetBundleNames(); // 获取所有打包文件
    foreach (string abName in allABNames)
    {
        string[] assetPaths = AssetDatabase.GetAssetPathsFromAssetBundle(abName); // 获取打包文件路径
        foreach (string path in assetPaths)
        {
            AssetImporter importer = AssetImporter.GetAtPath(path); // 获取导入器
            importer.assetBundleName = ""; // 清空打包名称
            importer.assetBundleVariant = ""; // 清空打包后缀
            EditorUtility.SetDirty(importer); // 保存导入器
        }
    }
    AssetDatabase.RemoveUnusedAssetBundleNames(); // 清除所有 AssetBundle 名称
```

## 加载流程
1. 加载 主 Manifest
``` cs
    string bundleDir = System.IO.Path.Combine(Application.dataPath, "..", "Bundles", buildTarget.ToString()); // 获取 自定义AssetBundle 目录
    string manifestPath = System.IO.Path.Combine(bundleDir, buildTarget.ToString()); // 获取 主 Manifest 路径
    AssetBundleManifest manifest = manifestBundle.LoadAsset<AssetBundleManifest>("AssetBundleManifest"); // 加载 主 Manifest
```

2. 加载依赖
``` cs
    string[] deps = manifest.GetAllDependencies(bundleName); // 获取依赖包

    foreach (string dep in deps)
    {
        string depPath = System.IO.Path.Combine(bundleDir, dep); // 获取依赖包路径
        AssetBundle depAB = AssetBundle.LoadFromFile(depPath); // 加载依赖包
        loadedDepBundles.Add(depAB);  // 追踪依赖包，方便实验时卸载
    }
```

3. 加载目标 AB 包
``` cs
    string targetPath = System.IO.Path.Combine(bundleDir, bundleName); // 获取目标包路径
    
    // 磁盘读取
    AssetBundleCreateRequest request = AssetBundle.LoadFromFileAsync(targetPath); // 加载目标包
    yield return request;
    targetAB = request.assetBundle; // 获取目标包

    // 内存读取
    AssetBundleCreateRequest memRequest = AssetBundle.LoadFromMemoryAsync(fileData); // 加载目标包
    yield return memRequest;
    targetAB = memRequest.assetBundle; // 获取目标包

    // 网络读取
    string fileUrl = "file://" + targetPath; // 获取目标包 URL
    using (UnityEngine.Networking.UnityWebRequest www =
        UnityEngine.Networking.UnityWebRequestAssetBundle.GetAssetBundle(fileUrl)) // 读取目标包
    {
        yield return www.SendWebRequest();
        if (www.result == UnityEngine.Networking.UnityWebRequest.Result.Success)
            targetAB = UnityEngine.Networking.DownloadHandlerAssetBundle.GetContent(www); // 获取目标包
    }
```

4. 加载实例化
``` cs
    GameObject prefab = targetAB.LoadAsset<GameObject>(prefabName); // 加载预制体
    GameObject instance = Instantiate(prefab, spawnPosition, Quaternion.identity); // 实例化预制体
```

5. 卸载流程
``` cs
    targetAB.Unload(false); // 卸载目标包
    manifestBundle.Unload(false); // 卸载清单包

    // 卸载依赖包 手动管理
    foreach (var depAB in loadedDepBundles)
    {
        depAB.Unload(true);  // true = 连资源一起销毁
    }
    loadedDepBundles.Clear();
```
